<script>
const screens = {
  start: document.getElementById('start'),
  bday: document.getElementById('bday'),
  ready: document.getElementById('ready'),
  intro: document.getElementById('intro'),
  game: document.getElementById('game'),
  word: document.getElementById('word'),
  giftBtnScreen: document.getElementById('giftBtnScreen'),
  gift: document.getElementById('gift')
};

function showScreen(next){
  Object.values(screens).forEach(s => s.classList.remove('active'));
  next.classList.add('active');
}

// CONFETTI only on Happy Birthday
function createConfetti(container){
  for(let i=0;i<80;i++){
    const c=document.createElement('div');
    c.className='confetti';
    c.style.left=Math.random()*100+'vw';
    c.style.backgroundColor=`hsl(${Math.random()*360},100%,50%)`;
    c.style.animationDuration=(2+Math.random()*2)+'s';
    container.appendChild(c);
    setTimeout(()=>c.remove(),4000);
  }
}

// FLOW sequential
document.getElementById('startBtn').onclick=()=>{
  showScreen(screens.bday);
  createConfetti(screens.bday);
  setTimeout(()=>showScreen(screens.ready),3000);
};

document.getElementById('readyBtn').onclick=()=>showScreen(screens.intro);
document.getElementById('gameBtn').onclick=()=>{ showScreen(screens.game); loadQuestion(); };

// QUIZ
const questions=[
  ["Who's birthday is it today?",["Si Tita","Kapitbahay","Hannah Rian Manuel Pascua my Loves"],2,"Yeheyyy Happy Birthday baby!! 🎉"],
  ["Sino mas cute?",["Ikaw","Ako","Parehas (pero mas lamang ikaw)"],2,"Of course parehas pero mas lamang ka pa din 😉"],
  ["What am I to you?",["Friend","Classmate","Favorite Person"],2,"Yeheyyy, ofc ako na 'to e great job baby!! 😍"]
];

let qIndex=0;

function loadQuestion(){
  const q=questions[qIndex];
  document.getElementById('qTitle').textContent=q[0];
  const box=document.getElementById('qOptions');
  box.innerHTML='';

  q[1].forEach((opt,i)=>{
    const btn=document.createElement('button');
    btn.textContent=opt;
    btn.onclick=()=>{
      if(i===q[2]){
        popup(q[3],()=>{
          qIndex++;
          if(qIndex<questions.length) loadQuestion();
          else showScreen(screens.word);
        });
      } else {
        popup("❌ Try again baby u can do it 😘");
      }
    };
    box.appendChild(btn);
  });
}

// GUESS THE WORD
document.getElementById('wordBtn').onclick=()=>{
  const val=document.getElementById('wordInput').value.toLowerCase();
  if(val==="maganda"){
    popup("Congrats to the most maganda and cutest na nilalang sa balat ng lupa!!! 😍",()=>showScreen(screens.giftBtnScreen));
  } else {
    popup("Try again baby!!");
  }
  document.getElementById('wordInput').value='';
};

// OPEN GIFT BUTTON → CAT STAGE
document.getElementById('giftBtn').onclick=showCatStage;

// CAT + LETTER
function showCatStage(){
  showScreen(screens.gift);
  const cat=document.getElementById('cat');
  cat.style.display='block';
  document.getElementById('letterPrompt').style.display='none';
  setTimeout(()=>{
    cat.style.display='none';
    document.getElementById('letterPrompt').style.display='flex';
  },2000); // letter appears 2s after cat
}

// POPUP
function popup(msg,cb){
  const p=document.getElementById('popup');
  document.getElementById('popupText').textContent=msg;
  p.style.display='block';
  document.getElementById('popupBtn').onclick=()=>{
    p.style.display='none';
    if(cb) cb();
  };
};
